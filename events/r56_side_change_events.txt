####### Event file for countries who might change sides in a war, IE; Romania / Finland

add_namespace = rt56.backstab_faction

# Warning Event
country_event = {
    id = rt56.backstab_faction.1
    title = rt56.backstab_faction.1.t
    desc = rt56.backstab_faction.1.d

    picture = GFX_report_event_generic_read_write
	
	is_triggered_only = yes
	
    option = {
        name = rt56.backstab_faction.1.a
		add_ai_strategy = {
			type = antagonize
			id = "FROM"			
			value = 300
		}
        add_political_power = 10                  
    }   
}

# We got stabbed in the back
country_event = {
    id = rt56.backstab_faction.2
    title = rt56.backstab_faction.2.t
    desc = rt56.backstab_faction.2.d

    picture = GFX_report_event_generic_read_write
	
	is_triggered_only = yes
	
    option = {
        name = rt56.backstab_faction.2.a
		add_ai_strategy = {
			type = antagonize
			id = "FROM"			
			value = 300
		}
        add_political_power = 10                  
    }   
}
# We want to stab them in the back
country_event = {
    id = rt56.backstab_faction.3
    title = rt56.backstab_faction.3.t
    desc = rt56.backstab_faction.3.d

    picture = GFX_report_event_generic_read_write
	
	is_triggered_only = yes
	
    option = {
        name = rt56.backstab_faction.3.a
        ai_chance = {
            factor = 2
            modifier = {
            }            
        }
        effect_tooltip = {
            any_state = {
                limit = {
                    is_owned_by = ROOT 
                    is_controlled_by = FROM
                }
                transfer_state = PREV
            }
            every_country = {
                limit = {
                    is_in_faction_with = ROOT
                    has_war_with = FROM
                }
                white_peace = FROM
            }
            if = {
                every_country = {
                    limit = {
                        is_in_faction_with = FROM
                        is_major = yes
                        is_faction_leader = yes
                        has_war_with = ROOT
                    }
                    FROM = {
                        leave_faction = yes
                    }
                    ROOT = {
                        add_to_faction = FROM
                    }
                }
            }
            if = {
                every_country = {
                    limit = {
                        has_war_with = ROOT
                        is_major = yes
                        is_faction_leader = yes
                        is_ai = yes
                    }
                    FROM = {
                        declare_war_on = {
                            target = PREV
                            type = topple_government
                        }
                    }
                }
            }
            else = {
                every_country = {
                    limit = {
                        has_war_with = ROOT
                        is_major = yes
                        is_faction_leader = yes
                        is_ai = no
                    }
                    FROM = {
                        create_wargoal = {
                            target = PREV
                            type = topple_government
                            expire = 0
                        }
                    }
                }                
            }
        }
        FROM = {
            country_event = { days = 2 id = rt56.backstab_faction.4 }
        }
    }
    option = {
        name = rt56.backstab_faction.3.b
        ai_chance = {
            factor = 2           
        }
        FROM = {
        }
    }    
}
