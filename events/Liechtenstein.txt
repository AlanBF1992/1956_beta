# Death of Franz I
country_event = {
    id = liechtenstein.1
    picture = GFX_report_event_europe_funeral
	title = liechtenstein.1.t
	desc = liechtenstein.1.d
    fire_only_once = yes

    is_triggered_only = yes

    trigger = {
        date > 1938.7.25
        has_country_leader = {
            character = LIE_franz_i
        }
    }

    mean_time_to_happen = {
        days = 20
    }

    option = {
        name = liechtenstein.1.a
        ai_chance = {
			base = 10
        }
        retire_character = LIE_franz_i
        add_country_leader_role = {
            character = LIE_franz_ii
            promote_leader = yes
            country_leader = {
                ideology = absolute_monarchism
                expire = "1965.1.1.1"
                traits = { }
            }
        }
    }
     
}

	title = liechtenstein.2.t
	desc = liechtenstein.2.d
    fire_only_once = yes

    is_triggered_only = yes

    option = { # Yes
        name = liechtenstein.2.a
        ai_chance = {
			base = 90
        }
        if = {
            limit = {
                ROOT = {
                    NOT = {
                        has_guaranteed = LIE
                    }
                }
                give_guarantee = LIE
            }
        }
        if = {
            limit = {
                LIE = {
                    NOT = {
                        gives_military_access_to = SWI
                    }
                }
                LIE = {
                    give_military_access = ROOT
                }
            }
        }
        if = {
            limit = {
                LIE = {
                    NOT = {
                        gives_military_access_to = SWI
                    }
                }
                LIE = {
                    give_military_access = ROOT
                }
            }
        }
        add_opinion_modifier = { target = LIE modifier = SWI_friendship }
        LIE = {
            country_event = { id = liechtenstein.3 hours = 6 }
        }
    }
    option = {
        name = liechtenstein.2.b
        ai_chance = {
			factor = 0
        }
        LIE = {
            country_event = { id = liechtenstein.4 hours = 6 }
        }
        
    }
}
