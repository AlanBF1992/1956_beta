				has_government = communism

			}
		}
		URG = {
			has_government = communism
		}
		SPR = { has_civil_war = yes }

			}

		NOT = { has_global_flag = scw_over }
	}

	abort = {
		OR = {
			date > 1939.1.1

		}
	}

	ai_strategy = {
		type = force_build_armies

		value = 1
	}

		type = befriend
