	allowed = { original_tag = SWE }

	enable = {
		has_war = no
		ROOT = {
			has_government = democratic
			NOT = { has_government = fascism }
			NOT = { is_in_faction_with = GER }
			NOT = { has_war_with = GER }
			is_in_faction = no
		}
		GER = { 
			exists = yes
			has_government = fascism 
			is_major = yes
		}
	}

	abort_when_not_enabled = yes	
	
	
	ai_strategy = {
		type = befriend
		id = "GER"
		value = 10
	}

	ai_strategy = {
		type = alliance
		id = "GER"
		value = -10
	}

	ai_strategy = {
		type = alliance
		id = "ENG"
		value = -10
	}

	ai_strategy = {
		type = send_lend_lease_desire
		id = "GER"
		value = -10
	}

	ai_strategy = {
		type = send_volunteers_desire
		id = "GER"
		value = -10
	}
}
