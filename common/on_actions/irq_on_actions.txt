on_actions = {
	on_startup = {
		effect = {
			IRQ = {
				if = {
					country_event = {
						id = iraq_hist.1
						days = 1552
					}
				}
			}
		}
	}

	on_capitulation = {
		effect = {
			if = {	### British peace offer
				limit = {
					tag = IRQ
					has_country_flag = IRQ_historical_golden_square_coup
				}
				IRQ = {
					country_event = { id = iraq_hist.5 }
				}
			}
		}
	}

	on_declare_war = {
		effect = {
			if = {
				limit = {
					tag = IRQ
					has_country_flag = IRQ_historical_golden_square_coup
				}
				hidden_effect = {
					every_country = {
						if = {
							limit = {
								exists = yes
								has_capitulated = no
								OR = {
									any_owned_state = {
										is_on_continent = 7
									}
									is_neighbor_of = IRQ
									has_war_with = GER
									has_war_with = ENG
									is_in_faction_with = GER
									is_in_faction_with = ENG
									is_major = yes
								}
							}
							news_event = { 
								hours = 4
								id = iraq_hist.3
							}
						}
					}
				}
			}
		}
	}

}
