	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = {
						is_in_array = {
							GLOBAL.r56_sa_muslim_states = THIS
						}
					}
				}
				if = {
					limit = {
						NOT = {
							original_tag = PAK
							original_tag = BAN
							original_tag = HYD
							original_tag = MLD
						}
					}
					if = {
						limit = { FROM.FROM = { is_core_of = ROOT } }
						add_dynamic_modifier = { modifier = r56_sa_religious_minority }
					}
					else = {
						add_dynamic_modifier = { modifier = r56_sa_religious_minority_occupied }
					}
				}

					limit = {
						is_in_array = {
							ROOT.slight_minority_states = FROM.FROM
						}

					}
					remove_dynamic_modifier = { modifier = r56_sa_religious_minority }
					add_dynamic_modifier = { modifier = r56_sa_slight_religious_minority }
				}
