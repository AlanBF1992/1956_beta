sa_remove_modifiers = {
	if = {
		limit = {
			has_dynamic_modifier = { modifier = r56_sa_religious_minority_occupied }
		}
		remove_dynamic_modifier = { modifier = r56_sa_religious_minority_occupied }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = { modifier = r56_sa_slight_religious_minority }
		}
		remove_dynamic_modifier = { modifier = r56_sa_slight_religious_minority }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = { modifier = r56_sa_religious_minority }
		}
		remove_dynamic_modifier = { modifier = r56_sa_religious_minority }
	}
}
sa_assign_muslim = {
	if = {
		limit = {
			PREV = {
				NOT = {
					original_tag = PAK
					original_tag = BAN
					original_tag = HYD
					original_tag = MLD
				}
			}
		}
		if = {
			limit = { is_core_of = PREV }
			add_dynamic_modifier = { 
				modifier = r56_sa_religious_minority 
			}
		}
		else = {
			add_dynamic_modifier = { 
				modifier = r56_sa_religious_minority_occupied
			}
		}
	}
	else_if = {
		limit = {
			is_in_array = {
				PREV.slight_minority_states = THIS
			}
			is_core_of = PREV
		}
		add_dynamic_modifier = { modifier = r56_sa_slight_religious_minority }
	}
}
