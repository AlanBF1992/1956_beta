ROM_increase_political_stability = {
	if = {
		limit = { has_idea = ROM_political_instability_2 }
		remove_ideas = ROM_political_instability_2
	}
	if = {
		limit = { has_idea = ROM_political_instability_1 }
		swap_ideas = {
			remove_idea = ROM_political_instability_1
			add_idea = ROM_political_instability_2
		}
	}
}
